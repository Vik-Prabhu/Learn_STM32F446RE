72void LED_Init(void)
73{
74 /* Step 1: Enable GPIOA clock */
75 RCC->AHB1ENR |= RCC_AHB1ENR_GPIOAEN;
76
77 /* Step 2: Set PA5 as output mode (MODER5 = 01) */
78 GPIOA->MODER &= ~(0x3U << (LED_PIN * 2)); /* Clear bits first */
79 GPIOA->MODER |= (0x1U << (LED_PIN * 2)); /* Set output mode */
80
81 /* Step 3: Set PA5 as push-pull output (OTYPER5 = 0) */
82 GPIOA->OTYPER &= ~(0x1U << LED_PIN);
83
84 /* Step 4: Set PA5 to low speed (OSPEEDR5 = 00) */
85 GPIOA->OSPEEDR &= ~(0x3U << (LED_PIN * 2));
86
87 /* Step 5: No pull-up no pull-down (PUPDR5 = 00) */
